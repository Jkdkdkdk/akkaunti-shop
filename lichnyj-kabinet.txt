        function showAuthForm(type) {
            document.getElementById('login-error').style.display = 'none';
            document.getElementById('register-error').style.display = 'none';
            document.getElementById('login-form').classList.remove('active');
            document.getElementById('register-form').classList.remove('active');
            document.getElementById(`${type}-form`).classList.add('active');
        }

        function updateAuthUI() {
            if (currentUser) {
                document.getElementById('auth-buttons').style.display = 'none';
                document.getElementById('user-menu').style.display = 'flex';
                document.getElementById('user-name').textContent = currentUser.name;
                document.querySelector('.admin-link').style.display = currentUser.isAdmin ? 'block' : 'none';
                document.getElementById('login-form').classList.remove('active');
                document.getElementById('register-form').classList.remove('active');
            } else {
                document.getElementById('auth-buttons').style.display = 'flex';
                document.getElementById('user-menu').style.display = 'none';
                document.querySelector('.admin-link').style.display = 'none';
            }
        }

        // ========== ОБНОВЛЕННЫЕ ФУНКЦИИ С ИСПОЛЬЗОВАНИЕМ API ==========

        async function login() {
            const email = document.getElementById('login-email').value;
            const password = document.getElementById('login-password').value;
            const errorElement = document.getElementById('login-error');
            
            const result = await apiCall('login', { email, password });
            
            if (result.error) {
                errorElement.textContent = result.error;
                errorElement.style.display = 'block';
            } else {
                currentUser = result.user;
                localStorage.setItem('currentUser', JSON.stringify(currentUser));
                updateAuthUI();
                document.getElementById('login-form').classList.remove('active');
                document.getElementById('login-email').value = '';
                document.getElementById('login-password').value = '';
                alert(`Добро пожаловать, ${result.user.name}!`);
            }
        }

        async function register() {
            const name = document.getElementById('register-name').value;
            const email = document.getElementById('register-email').value;
            const password = document.getElementById('register-password').value;
            const passwordConfirm = document.getElementById('register-password-confirm').value;
            const errorElement = document.getElementById('register-error');
            
            if (password !== passwordConfirm) {
                errorElement.textContent = 'Пароли не совпадают!';
                errorElement.style.display = 'block';
                return;
            }
            
            const result = await apiCall('register', { name, email, password });
            
            if (result.error) {
                errorElement.textContent = result.error;
                errorElement.style.display = 'block';
            } else {
                currentUser = result.user;
                localStorage.setItem('currentUser', JSON.stringify(currentUser));
                updateAuthUI();
                
                document.getElementById('register-form').classList.remove('active');
                alert('Регистрация успешна!');
                
                document.getElementById('register-name').value = '';
                document.getElementById('register-email').value = '';
                document.getElementById('register-password').value = '';
                document.getElementById('register-password-confirm').value = '';
            }
        }

        function logout() {
            currentUser = null;
            localStorage.removeItem('currentUser');
            updateAuthUI();
            alert('Вы вышли из аккаунта');
        }

        function showPage(page) {
            // Останавливаем автообновление чата при переходе на другие страницы
            if (chatRefreshInterval && page !== 'chat') {
                clearInterval(chatRefreshInterval);
                chatRefreshInterval = null;
            }
            
            document.querySelectorAll('.main-content, .cart-section, .payment-section, .chat-section, .admin-section').forEach(el => {
                el.style.display = 'none';
            });
            
            switch(page) {
                case 'home': 
                    document.getElementById('home-page').style.display = 'block'; 
                    break;
                case 'products': 
                    if (!currentUser) { 
                        alert('Войдите в аккаунт!'); 
                        showAuthForm('login');
                        return; 
                    }
                    document.getElementById('products-page').style.display = 'block'; 
                    break;
                case 'cart': 
                    if (!currentUser) { 
                        alert('Войдите в аккаунт!'); 
                        showAuthForm('login');
                        return; 
                    }
                    document.getElementById('cart-page').style.display = 'block'; 
                    updateCartUI();
                    break;
                case 'orders': 
                    if (!currentUser) { 
                        alert('Войдите в аккаунт!'); 
                        showAuthForm('login');
                        return; 
                    }
                    document.getElementById('orders-page').style.display = 'block'; 
                    loadOrdersHistory();
                    break;
                case 'admin': 
                    if (!currentUser || !currentUser.isAdmin) { 
                        alert('Нет доступа!'); 
                        return; 
                    }
                    document.getElementById('admin-page').style.display = 'block'; 
                    loadAdminData();
                    break;
                case 'payment':
                    document.getElementById('payment-page').style.display = 'block';
                    break;
                case 'chat':
                    document.getElementById('chat-page').style.display = 'block';
                    if (currentChatId) {
                        loadChatMessages(currentChatId);
                    }
                    break;
                default: 
                    document.getElementById('home-page').style.display = 'block';
            }
        }

        function loadProducts(category) {
            const container = document.getElementById('products-container');
            const filtered = products.filter(p => p.category === category);
            
            container.innerHTML = filtered.map(product => `
                <div class="product-card">
                    <div class="product-image">
                        <i class="fab fa-${product.category === 'vk' ? 'vk' : 'telegram'}"></i>
                        <div class="country-flag">${product.country}</div>
                        <div class="account-type">${product.type}</div>
                    </div>
                    <div class="product-info">
                        <div class="product-title">${product.name}</div>
                        <div class="product-description">${product.description}</div>
                        <div class="product-price">${product.price} ₽</div>
                        <button class="add-to-cart" data-id="${product.id}"><i class="fas fa-cart-plus"></i> В корзину</button>
                    </div>
                </div>
            `).join('');
            
            container.querySelectorAll('.add-to-cart').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    if (!currentUser) { 
                        alert('Войдите в аккаунт!'); 
                        showAuthForm('login');
                        return; 
                    }
                    const productId = e.target.closest('.add-to-cart').dataset.id;
                    const product = products.find(p => p.id == productId);
                    addToCart(product);
                    alert(`"${product.name}" добавлен в корзину!`);
                });
            });
        }

        function addToCart(product) {
            const existing = cart.find(item => item.id === product.id);
            if (existing) {
                existing.quantity += 1;
            } else {
                cart.push({ id: product.id, name: product.name, price: product.price, quantity: 1 });
            }
            localStorage.setItem('cart', JSON.stringify(cart));
            updateCartUI();
        }

        function updateCartUI() {
            const container = document.getElementById('cart-items');
            const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
            const itemCount = cart.reduce((sum, item) => sum + item.quantity, 0);
            
            document.getElementById('cart-badge').textContent = itemCount;
            document.getElementById('cart-badge').style.display = itemCount > 0 ? 'flex' : 'none';
            document.getElementById('cart-total').textContent = `${total} ₽`;
            
            if (cart.length === 0) {
                container.innerHTML = '<div class="empty-state"><i class="fas fa-shopping-cart"></i><h3>Корзина пуста</h3></div>';
                document.getElementById('checkout-btn').disabled = true;
                return;
            }
            
            document.getElementById('checkout-btn').disabled = false;
            container.innerHTML = cart.map(item => `
                <div class="cart-item">
                    <div class="cart-item-details">
                        <div class="cart-item-title">${item.name}</div>
                        <div class="cart-item-price">${item.price} ₽ x ${item.quantity} = ${item.price * item.quantity} ₽</div>
                    </div>
                    <div class="cart-item-actions">
                        <div class="quantity-control">
                            <button class="quantity-btn decrease" data-id="${item.id}">-</button>
                            <span class="quantity">${item.quantity}</span>
                            <button class="quantity-btn increase" data-id="${item.id}">+</button>
                        </div>
                        <button class="remove-item" data-id="${item.id}">&times;</button>
                    </div>
                </div>
            `).join('');
            
            container.querySelectorAll('.decrease').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const id = parseInt(e.target.dataset.id);
                    const item = cart.find(item => item.id === id);
                    if (item && item.quantity > 1) {
                        item.quantity -= 1;
                    } else {
                        cart = cart.filter(item => item.id !== id);
                    }
                    localStorage.setItem('cart', JSON.stringify(cart));
                    updateCartUI();
                });
            });
            
            container.querySelectorAll('.increase').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const id = parseInt(e.target.dataset.id);
                    const item = cart.find(item => item.id === id);
                    if (item) {
                        item.quantity += 1;
                        localStorage.setItem('cart', JSON.stringify(cart));
                        updateCartUI();
                    }
                });
            });
            
            container.querySelectorAll('.remove-item').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const id = parseInt(e.target.dataset.id);
                    cart = cart.filter(item => item.id !== id);
                    localStorage.setItem('cart', JSON.stringify(cart));
                    updateCartUI();
                });
            });
        }

        // ОБНОВЛЕННАЯ ФУНКЦИЯ ЗАГРУЗКИ ЧАТА С АВТООБНОВЛЕНИЕМ
        async function loadChatMessages(chatId) {
            const result = await apiCall('getChat', null, chatId);
            
            const messages = Array.isArray(result) ? result : [];
            const chatMessages = document.getElementById('chat-messages');
            
            chatMessages.innerHTML = messages.map(msg => {
                let senderClass = 'user';
                let senderName = 'Вы';
                
                if (msg.sender === 'manager') {
                    senderClass = 'manager';
                    senderName = 'Менеджер';
                } else if (msg.sender === 'admin') {
                    senderClass = 'admin';
                    senderName = 'Администратор';
                }
                
                return `
                <div class="message ${senderClass}">
                    <div class="message-sender">${senderName}</div>
                    <div class="message-text">${msg.text}</div>
                    ${msg.image ? `<img src="${msg.image}" class="message-image" alt="Изображение">` : ''}
                    <div class="message-time">${msg.time}</div>
                </div>
                `;
            }).join('');
            
            chatMessages.scrollTop = chatMessages.scrollHeight;
            
            // Запускаем автообновление чата только если мы на странице чата
            if (document.getElementById('chat-page').style.display !== 'none') {
                startChatAutoRefresh(chatId);
            }
        }

        // Функция автообновления чата
        function startChatAutoRefresh(chatId) {
            // Останавливаем предыдущий интервал
            if (chatRefreshInterval) {
                clearInterval(chatRefreshInterval);
            }
            
            // Запускаем новый интервал обновления каждые 3 секунды
            chatRefreshInterval = setInterval(async () => {
                // Обновляем только если чат открыт
                if (document.getElementById('chat-page').style.display !== 'none' && currentChatId === chatId) {
                    await refreshChatMessages(chatId);
                } else {
                    // Если чат закрыт, останавливаем обновление
                    clearInterval(chatRefreshInterval);
                    chatRefreshInterval = null;
                }
            }, 3000);
        }

        // Функция обновления сообщений без полной перезагрузки
        async function refreshChatMessages(chatId) {
            try {
                const result = await apiCall('getChat', null, chatId);
                const messages = Array.isArray(result) ? result : [];
                const chatMessages = document.getElementById('chat-messages');
                
                // Получаем текущее количество сообщений
                const currentMessageCount = chatMessages.querySelectorAll('.message').length;
                
                // Если появились новые сообщения, обновляем чат
                if (messages.length > currentMessageCount) {
                    chatMessages.innerHTML = messages.map(msg => {
                        let senderClass = 'user';
                        let senderName = 'Вы';
                        
                        if (msg.sender === 'manager') {
                            senderClass = 'manager';
                            senderName = 'Менеджер';
                        } else if (msg.sender === 'admin') {
                            senderClass = 'admin';
                            senderName = 'Администратор';
                        }
                        
                        return `
                        <div class="message ${senderClass}">
                            <div class="message-sender">${senderName}</div>
                            <div class="message-text">${msg.text}</div>
                            ${msg.image ? `<img src="${msg.image}" class="message-image" alt="Изображение">` : ''}
                            <div class="message-time">${msg.time}</div>
                        </div>
                        `;
                    }).join('');
                    
                    chatMessages.scrollTop = chatMessages.scrollHeight;
                }
            } catch (error) {
                console.error('Ошибка обновления чата:', error);
            }
        }

        // ИСПРАВЛЕННАЯ ФУНКЦИЯ ОТПРАВКИ СООБЩЕНИЙ
        async function sendMessage() {
            const input = document.getElementById('chat-input');
            const text = input.value.trim();
            
            if (!text || !currentChatId) return;
            
            // Блокируем кнопку отправки
            const sendButton = document.getElementById('send-message');
            sendButton.disabled = true;
            sendButton.textContent = 'Отправка...';
            
            try {
                // Получаем текущие сообщения
                const currentMessages = await apiCall('getChat', null, currentChatId);
                const messages = Array.isArray(currentMessages) ? currentMessages : [];
                
                // Определяем тип отправителя
                let senderType = 'user';
                if (currentUser.isAdmin) {
                    senderType = 'admin';
                }
                
                const newMessage = {
                    sender: senderType,
                    text: text,
                    time: new Date().toLocaleString('ru-RU'),
                    id: Date.now()
                };
                
                messages.push(newMessage);
                
                // Сохраняем обновленный чат
                await apiCall('updateChat', {
                    chatId: currentChatId,
                    messages: messages
                });
                
                input.value = '';
                await loadChatMessages(currentChatId);
                
                // Если это не админ, имитируем ответ менеджера
                if (!currentUser.isAdmin) {
                    setTimeout(async () => {
                        const responses = [
                            "Спасибо за сообщение! Проверяю ваш заказ...",
                            "Получил ваше сообщение, скоро отвечу!",
                            "Скриншот получен, проверяю оплату...",
                            "Ваш заказ обрабатывается, скоро пришлю данные!"
                        ];
                        
                        const randomResponse = responses[Math.floor(Math.random() * responses.length)];
                        
                        const managerMessage = {
                            sender: 'manager',
                            text: randomResponse,
                            time: new Date().toLocaleString('ru-RU'),
                            id: Date.now() + 1
                        };
                        
                        // Получаем текущие сообщения
                        const currentMessages = await apiCall('getChat', null, currentChatId);
                        const updatedMessages = Array.isArray(currentMessages) ? currentMessages : [];
                        
                        // Добавляем ответ менеджера
                        updatedMessages.push(managerMessage);
                        
                        // Сохраняем обновленный чат
                        await apiCall('updateChat', {
                            chatId: currentChatId,
                            messages: updatedMessages
                        });
                        
                        await loadChatMessages(currentChatId);
                    }, 1000); // Уменьшена задержка с 2000 до 1000 мс
                }
            } catch (error) {
                console.error('Ошибка отправки сообщения:', error);
                alert('Ошибка отправки сообщения. Попробуйте еще раз.');
            } finally {
                // Разблокируем кнопку отправки
                sendButton.disabled = false;
                sendButton.textContent = 'Отправить';
            }
        }
